# 一、MySQL 基础

  为了使用 mysql，可以单独为电脑安装 mysql 数据库 + Navicat 可视化工具，当然也可以直接使用一些集成环境（如：phpstudy、wampserver）。

  • mysql 安装版 (opens new window) 或者 解压版 (opens new window) + Navicat 可视化（默认端口号：3306）
  • wampserver 集成工具（默认端口号：80）

若使 mysql 解压版，关于环境变量就暂时不用配置了，直接在 mysql 数据库的 bin 目录打开终端操作即可。

① 启动服务端（mongod.exe）：

准备工作：

  • 配置环境变量（可以省略）

# 3、创建 mysql 模型

  • 新建 models / db.js
const mysql2 = require('mysql2')

const dbConfig = require('../config/db.config')

// 创建🚩连接池对象
const promisePool = mysql2
  .createPool({
    host: dbConfig.host,
    user: dbConfig.user,
    password: dbConfig.password,
    database: dbConfig.database
  })
  .promise()

// 向外共享 promisePool 数据库连接对象
module.exports = promisePool

  • services / userService.js
const promisePool = require('../models/db')

const userService = {
  addUser: async (username) => {
    const [rows, fields] = await promisePool
      .query(userModel.addUser, [username])
      .catch((err) => {
        return res.send({ code: 1, message: err.message })
      })

    console.log(rows) // 等效于✨下面的 users[0]

    if (rows.length > 0) {
      return res.send({ code: 1, message: '用户名被占用，请更换其他用户名！' })
    } else {
      return res.send({
        code: 0,
        message: '注册成功！'
      })
    }
  }
}

module.exports = userService
